# Ubuntu System Backup Cron Jobs
# This replaces the old snapshot configuration

# Configuration snapshots - daily at 2:00 AM
0 2 * * * /home/joey/config_snapshot.sh >> /var/log/config_snapshot.log 2>&1

# Full system backup - weekly on Sunday at 3:00 AM  
0 3 * * 0 /home/joey/system_backup.sh >> /var/log/system_backup.log 2>&1

# Home directory snapshot - daily at 1:00 AM (if you want to backup home separately)
# 0 1 * * * /usr/bin/btrfs subvolume snapshot -r /home "/mnt/snapshots/home/home_$(date +\%Y-\%m-\%d_\%H-\%M-\%S)" >> /var/log/home_snapshot.log 2>&1
